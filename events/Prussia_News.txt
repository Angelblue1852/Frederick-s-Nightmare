	major = yes

		name = GER_news.1.o1
		trigger = { original_tag = PRS }
	}

		name = GER_news.1.o2

			NOT = { original_tag = PRS }

				is_in_faction_with = PRS

			}
		}
	}
	option = {
		name = GER_news.1.o3
		trigger = { 
			NOT = { original_tag = PRS }
			NOT = { original_tag = SOV }

			NOT = { has_war_with = PRS }
			NOT = { has_opinion = { target = PRS value < -30 } }
			NOT = { has_opinion = { target = PRS value > 30 } }
		}
	}
	option = {
		name = GER_news.1.o4
		trigger = { 
			NOT = { original_tag = PRS }
			NOT = { is_in_faction_with = PRS }
			OR = {

		name = GER_news.1.o5
